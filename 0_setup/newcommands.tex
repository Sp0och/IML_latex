% mathbox around math statements
% PRE: #1 math expression
    \newcommand{\mathbox}[1]{
        \vskip3pt
        \begin{center}
            \boxed{
                #1
            }
        \end{center}
        \vskip4pt
    }

% box around text 4col
% PRE: #1 text
% Additional comment: 0.2345 is a hardcoded relation which holds
%                     only for a four-column enviroment. (pbox auto-shrinks)
\newcommand{\textbox}[1]{
    \vskip4pt
    \begin{center}
        \fbox{\pbox{0.2345\textwidth}{
            #1
        }}
    \end{center}
    \vskip4pt
}
    
% boxed picture
% PRE: #1(opt) picture size in relation to \linewidth, in (0,1], default: 0.9
%      #2 absolute picture path
    \newcommand{\boxGraphic}[2][0.9]{
        {
        \vskip4pt
        \begin{center}
        \setlength{\fboxsep}{0pt}%
        \setlength{\fboxrule}{0.2pt}%
        \fbox{\includegraphics[width={#1}\linewidth]{#2}}
        \end{center}
        \vskip4pt
        }
    }
